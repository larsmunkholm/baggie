# NonEmptyArray

Require an array to have at least one value.

> This type comes from here: https://stackoverflow.com/a/56006703/1446188

## Example

<Source dark language="ts" code={`
import { NonEmptyArray } from "@baggie/core";

type ArrayWithNumbers = NonEmptyArray<number>;

// ✅ Good
const arr: ArrayWithNumbers = [1, 2, 3];

// ✅ Good
const arr: ArrayWithNumbers = [1];

// ⛔️ TypeScript error
const arr: ArrayWithNumbers = [];
`} />
